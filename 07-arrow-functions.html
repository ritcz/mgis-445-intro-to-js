Arrow Functions

        <script>
          // Conventional function
          window.onload = () => {
            function calculateInterest(principal, rate, time) {
              const accruedAmount = principal * (1 + rate * time);
              return accruedAmount;
            }

            const amount = calculateInterest(30000, 0.03, 20);
          };

          // Arrow functions
          window.onload = () => {
            // with explicit return
            const calculateInterest = (principal, rate, time) => {
              const accruedAmount = principal * (1 + rate * time);
              return accruedAmount;
            };

            // with implicit return
            const calcWithImplicitReturn = (principal, rate, time) =>
              principal * (1 + rate * time);

            const amount = calculateInterest(30000, 0.03, 20);
            const amount2 = calcWithImplicitReturn(30000, 0.03, 20);
          };
        </script>
