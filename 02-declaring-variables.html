Declaring Variables

      <script>
        // changeable, legacy
        var x = 1;
        console.log("x", x);

        // changeable, modern
        let y = 2;
        console.log("y", y);

        // not changeable
        const z = 3;
        console.log("z", z);

        // change works
        x = 100;
        console.log("x", x);

        // change works
        y = 200;
        console.log("y", y);

        // Throws an error
        z = 300;
        console.log("z", z);
      </script>

Strict Mode

<script> "use strict"; // strict mode is on // no error because we declared the cow variable const cow = "better than pig"; console.log("cow", cow); // will cause an error and execution will stop horse = "better than cow"; console.log("horse", horse); console.log('but my cow!', cow); </script> <script> // this will still run - not in strict mode in this block donkey = "better than snake"; console.log("donkey", donkey); </script>
