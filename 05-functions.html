      <script>

        console.warn("Warning - This should be avoided but works!");

        // query the document
        const theButton = document.querySelector("#the-button");
        console.log("The Button", theButton);

        const theDivs = document.querySelectorAll(".sample");
        console.log("The Sample Divs", theDivs);

        // get unix timestamp
        const now = Date.now();
        console.log(`now: ${now}`);

        // parse string into an integer
        const intValue = parseInt("20");
        console.log(`intValue: ${intValue}`);

        // parse string into a float (number)
        const floatString = "20.232";
        const floatValue = parseFloat(floatString);
        console.log(`float value: ${floatValue}, type ${typeof floatValue}`);
