Arrays

Declaration, Types, Access

      <script>
        // Declare an array
        const fruits = ["apple", "banana", "cherry"];

        // Delcare an object
        const trees = { maple: "maple", oak: "oak", cedar: "cedar" };

        // show that both are objects
        console.log(`Type of trees: ${typeof trees}`);
        console.log(`Type of fruits: ${typeof fruits}`);

        // shows that only fruit is an array
        console.log(`Are trees an array? ${Array.isArray(trees)}`);
        console.log(`Are fruits an array? ${Array.isArray(fruits)}`);

        // Access an item in the array
        console.log(`The first element in the array is "${fruits[0]}".`);

        // Access the length of the array using the length property
        console.log(`There are ${fruits.length} elements in the array`);

        // Iterate through an array
        for (let i = 0; i < fruits.length; i++) {
          const fruit = fruits[i];
          console.log(`${fruit} is element ${i}.`);
        }
      </script>

Array Methods

      <script>
        // Add an element to the end of the array with push()
        fruits.push("dragon fruit");
        console.log("Pushed an element", fruits);

        // Add an element to the beginning of the array with unshift()
        fruits.unshift("acai berry");
        console.log("Unshifted an element", fruits);

        const lastFruit = fruits.pop();
        console.log(`Popped ${lastFruit} from fruits.`);
        console.log("Remaining fruits:", fruits);

        // Sort elements an array
        const numbers = [10, 27, 30, 12, 87, 52, 4];
        console.log("Before Sorting", numbers);

        // note that this actually changes the array
        numbers.sort((a, b) => a - b);
        console.log("After Sorting", numbers);

        // reverses the order of the elements in the array
        numbers.reverse();
        console.log("After reversing", numbers);

        const luckyNumbers = numbers.slice(2, 5);
        console.log("Slice of the array", luckyNumbers);
      </script>
